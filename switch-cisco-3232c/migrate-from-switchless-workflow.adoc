:icons: font
:imagesdir: ../media/


[role="lead"]

.image:https://raw.githubusercontent.com/NetAppDocs/common/main/media/number-3.png["Drittens"]link:migrate-from-switchless-configure-ports.html["Konfigurieren Sie Ihre Ports"]
